const SUPABASE = "https://ezektgzwesrtezeghmrs.supabase.co";
const ORIGIN   = "https://mhmmedjalaom-glitch.github.io";
const CORS = {
  "Access-Control-Allow-Origin": ORIGIN,
  "Access-Control-Allow-Methods": "GET,POST,PUT,PATCH,DELETE,OPTIONS",
  "Access-Control-Allow-Headers": "apikey,authorization,content-type,prefer,accept",
  "Access-Control-Max-Age": "86400",
};
export default {
  async fetch(req) {
    if (req.method === "OPTIONS")
      return new Response(null, { status: 204, headers: CORS });
    try {
      const u = new URL(req.url);
      const h = new Headers(req.headers);
      h.delete("host");
      const body = ["GET","HEAD"].includes(req.method) ? null : req.body;
      const res = await fetch(SUPABASE + u.pathname + u.search,
        { method: req.method, headers: h, body });
      const r = new Response(res.body, res);
      Object.entries(CORS).forEach(([k,v]) => r.headers.set(k,v));
      return r;
    } catch(e) {
      return new Response(JSON.stringify({error:e.message}),
        {status:502, headers:{"Content-Type":"application/json",...CORS}});
    }
  }
};
